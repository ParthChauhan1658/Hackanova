  'use client'
  import { useState, useEffect } from 'react'
  import { motion, AnimatePresence } from 'framer-motion'
  import { useVitalsStore } from '@/store/vitals-store'
  import { PersonStanding, CheckCircle2, Timer } from 'lucide-react'
  import { sentinel } from '@/lib/sentinel-api'

  export function FallOverlay() {
    const { currentVitals, activePatientId } = useVitalsStore()
    const [visible, setVisible] = useState(false)
    const [countdown, setCountdown] = useState(60)
    const [acknowledged, setAcknowledged] = useState(false)

    useEffect(() => {
      if (currentVitals?.fallEvent === 'POSSIBLE_FALL' || currentVitals?.fallEvent === 'CONFIRMED_FALL') {
        setVisible(true); setCountdown(60); setAcknowledged(false)
      }
    }, [currentVitals?.fallEvent])

    useEffect(() => {
      if (!visible || acknowledged) return
      const t = setInterval(() => setCountdown(c => { if (c <= 1) { clearInterval(t); return 0 } return c - 1 }), 1000)
      return () => clearInterval(t)
    }, [visible, acknowledged])

    const handleAcknowledge = async () => {
      if (activePatientId) await sentinel.acknowledgeFall(activePatientId)
      setAcknowledged(true)
      setTimeout(() => setVisible(false), 2000)
    }

    return (
      <AnimatePresence>
        {visible && (
          <motion.div
            initial={{ opacity: 0, x: '100%' }}
            animate={{ opacity: 1, x: 0 }}
            exit={{ opacity: 0, x: '100%' }}
            transition={{ type: 'spring', stiffness: 280, damping: 28 }}
            className="fixed bottom-6 right-6 z-[90] w-80 rounded-3xl border border-[oklch(0.65_0.18_240_/_0.4)] bg-[oklch(0.20_0.06_240_/_0.92)]       
  backdrop-blur-2xl p-5 shadow-[0_0_40px_-8px_oklch(0.65_0.18_240)]"
          >
            <div className="flex items-center gap-3 mb-3">
              <motion.div animate={{ rotate: [0, -10, 10, 0] }} transition={{ duration: 0.6, repeat: 3 }}>
                <PersonStanding className="w-7 h-7 text-[oklch(0.65_0.18_240)]" />
              </motion.div>
              <div>
                <p className="font-bold text-white text-sm">Fall Detected</p>
                <p className="text-xs text-[oklch(0.85_0.06_100)]">
                  {currentVitals?.fallEvent === 'CONFIRMED_FALL' ? 'CONFIRMED — EMS pending' : 'Possible fall — monitoring'}
                </p>
              </div>
            </div>

            {!acknowledged && (
              <div className="mb-3">
                <div className="flex justify-between text-xs text-[oklch(0.85_0.06_100)] mb-1">
                  <span className="flex items-center gap-1"><Timer className="w-3 h-3" /> EMS in</span>
                  <span className={countdown < 15 ? 'text-[oklch(0.55_0.25_25)] font-bold' : ''}>{countdown}s</span>
                </div>
                <div className="h-1.5 rounded-full bg-[oklch(1_0_0_/_0.1)] overflow-hidden">
                  <motion.div
                    className="h-full rounded-full bg-[oklch(0.65_0.18_240)]"
                    animate={{ width: `${(countdown / 60) * 100}%` }}
                    transition={{ duration: 1 }}
                  />
                </div>
              </div>
            )}

            {acknowledged
              ? (
                <motion.div initial={{ scale: 0.8 }} animate={{ scale: 1 }} className="flex items-center gap-2 text-[oklch(0.72_0.17_145)] text-sm      
  font-semibold">
                  <CheckCircle2 className="w-5 h-5" /> Acknowledged — EMS cancelled
                </motion.div>
              ) : (
                <motion.button
                  whileHover={{ scale: 1.03 }} whileTap={{ scale: 0.97 }}
                  onClick={handleAcknowledge}
                  className="w-full py-2.5 rounded-xl bg-[oklch(0.65_0.18_240_/_0.4)] border border-[oklch(0.65_0.18_240_/_0.5)] text-white font-bold   
  text-sm"
                >
                  I'm OK — Cancel EMS
                </motion.button>
              )
            }
          </motion.div>
        )}
      </AnimatePresence>
    )
  }
